include(Catch)

function(yac_add_test name source)
  add_executable(${name} ${source})
  target_link_libraries(${name} PRIVATE
    yac_presentation
    yac_service
    Catch2::Catch2WithMain
  )
  catch_discover_tests(${name})
endfunction()
